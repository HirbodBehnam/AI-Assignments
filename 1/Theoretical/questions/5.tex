\begin{gather*}
    \frac{df}{dx} = 2x - y\\
    \frac{df}{dy} = 4y - x\\
    \nabla f = \begin{pmatrix}
        2x - y\\
        4y - x
    \end{pmatrix} 
\end{gather*}
